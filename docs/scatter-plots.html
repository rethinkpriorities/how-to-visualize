5 Scatter plots

5.1 Simple scatter plot 

p <- ggplot(mtcars, aes(x = wt, y = mpg)) +
  geom_point(alpha = .85) +
  theme_rethink()

ggplotly(p)

5.2 Jittered scatter plot 

p <- ggplot(mtcars, aes(x = wt, y = mpg)) +
  geom_jitter(alpha = .85, width = 0.5, height = 0.5) +
  theme_rethink()

ggplotly(p, tooltip = "none")

5.3 Multiple scatter plot 

p <- ggplot(mtcars, aes(x = wt, y = mpg, color = factor(cyl))) +
  geom_point(alpha = .85) +
  scale_color_viridis(discrete = TRUE, option = "mako", end = .95) +
  theme_rethink()

ggplotly(p)
